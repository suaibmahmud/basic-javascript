// global & local scope

var num = 10; // global variable
function func1(){
    var num1 = 5; // local variable 
}

function func2(){
    var num = 15; // num will be unchanged because num is a global variable, it cannot be reasigned in function 
    console.log(num); // correct
    console.log(num1); // error
}

func2();
